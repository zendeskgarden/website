### Tokens

Icon tokens represent specific concepts in Zendesk products.
Before adding an icon to your project, check the list below to see if the
concept has an icon token associated with it.

export const pageQuery = graphql`
  query ($fileAbsolutePath: String) {
    ...SidebarPageFragment
    svgIcons: allFile(
      filter: {
        sourceInstanceName: { eq: "svg-icons" }
        relativeDirectory: { eq: "16" }
        name: { glob: "!wordmark-*" }
      }
      sort: { fields: name, order: ASC }
    ) {
      totalCount
      edges {
        node {
          name
          relativeDirectory
          childGardenSvg {
            content
          }
        }
      }
    }
    tokens: allIconToken {
      totalCount
      edges {
        node {
          style
          synonyms
          icon
          id
        }
      }
    }
  }
`;
