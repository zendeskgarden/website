### Close

Tags can be dismissed.

import TagsClose from '../../examples/tags/TagsClose.tsx';
import TagsCloseCode from '!!raw-loader!../../examples/tags/TagsClose.tsx';

<CodeExample code={TagsCloseCode}>

### Color

Tags can be colored using a [theme palette](/components/theme-object#palette)
hue or hex value.

import TagsColor from '../../examples/tags/TagsColor.tsx';
import TagsColorCode from '!!raw-loader!../../examples/tags/TagsColor.tsx';

<CodeExample code={TagsColorCode}>
  <TagsColor />
</CodeExample>

### Shape

Tags are rectangular by default. Other options are pill and round. Round tags
are reserved for very short content like two digit numbers.

import TagsShape from '../../examples/tags/TagsShape.tsx';
import TagsShapeCode from '!!raw-loader!../../examples/tags/TagsShape.tsx';

<CodeExample code={TagsShapeCode}>
  <TagsShape />
</CodeExample>

### Size

Tags can be small, medium, or large. They are medium by default.

import TagsSize from '../../examples/tags/TagsSize.tsx';
import TagsSizeCode from '!!raw-loader!../../examples/tags/TagsSize.tsx';

<CodeExample code={TagsSizeCode}>
  <TagsSize />
</CodeExample>

### Regular weight

Tag content is bold by default, but can be regular weight too.

import TagsWeight from '../../examples/tags/TagsWeight.tsx';
import TagsWeightCode from '!!raw-loader!../../examples/tags/TagsWeight.tsx';
