<Usage>
  <Use>
    <ul>
      <li>To let users compare options from a list and select all, any, or none of those items</li>
      <li>To turn a single option on or off</li>
    </ul>
  </Use>
  <Misuse>
    <ul>
      <li>
        To give the user a mutually exclusive choice use the <a href="./radio">Radio</a> component
        instead
      </li>
      <li>
        To let users activate an option that takes effect immediately use a{' '}
        <a href="./toggle">Toggle</a> instead
      </li>
    </ul>
  </Misuse>
</Usage>
